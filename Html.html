document.addEventListener("DOMContentLoaded", function() { // Array of options for the dropdown const options = ["Option 1", "Option 2", "Option 3", "Option 4"]; // Create select element const select = document.createElement("select"); select.setAttribute("id", "dynamicSelect"); // Populate select element with options options.forEach(option => { const opt = document.createElement("option"); opt.value = option; opt.innerHTML = option; select.appendChild(opt); }); // Append select element to the dropdown container document.getElementById("dropdownContainer").appendChild(select); // Add event listener for the select element select.addEventListener("change", function() { alert("You selected: " + this.value); }); });
